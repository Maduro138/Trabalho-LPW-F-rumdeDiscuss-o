Exemplos de Programas em C

1. Olá Mundo

#include <stdio.h>

int main() {
    printf("Olá, Mundo!\n");
    return 0;
}

2. Soma de Dois Números

#include <stdio.h>

int main() {
    int a, b;
    printf("Digite dois números: ");
    scanf("%d %d", &a, &b);

    int soma = a + b;
    printf("Soma = %d\n", soma);

    return 0;
}

3. Maior Número

#include <stdio.h>

int main() {
    int x, y;
    printf("Digite dois valores: ");
    scanf("%d %d", &x, &y);

    if (x > y) {
        printf("Maior = %d\n", x);
    } else {
        printf("Maior = %d\n", y);
    }

    return 0;
}

4. Contagem com For

#include <stdio.h>

int main() {
    for(int i = 1; i <= 10; i++) {
        printf("%d\n", i);
    }
    return 0;
}

5. Média de Números

#include <stdio.h>

int main() {
    float a, b, c;
    printf("Digite três notas: ");
    scanf("%f %f %f", &a, &b, &c);

    float media = (a + b + c) / 3;
    printf("Média = %.2f\n", media);

    return 0;
}

6. Vetor Simples

#include <stdio.h>

int main() {
    int v[5] = {10, 20, 30, 40, 50};

    for(int i = 0; i < 5; i++) {
        printf("%d ", v[i]);
    }

    return 0;
}
